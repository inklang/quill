# Ink LSP Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a VS Code extension with an LSP server that provides syntax highlighting, autocomplete, diagnostics, and hover for the Ink language — dynamically adapting based on installed grammar packages.

**Architecture:** Separate repo (`ink-lsp`). Two bundles: extension host (starts server via stdio) and LSP server (child process). Three-layer highlighting: static TextMate grammar for base keywords, semantic tokens for grammar-package keywords, and LSP intelligence for completions/diagnostics/hover. Grammar context loaded from `ink-package.toml` + `quill.lock` + `grammar.ir.json` files.

**Tech Stack:** TypeScript, esbuild, vscode-languageclient/server, @iarna/toml, vitest

**Spec:** `docs/superpowers/specs/2026-03-28-ink-lsp-design.md` (in quill repo)

```ts
// Grammar IR types — mirrors quill/src/grammar/ir.ts

export type Rule =
  | { type: 'seq'; items: Rule[] }
  | { type: 'choice'; items: Rule[] }
  | { type: 'many'; item: Rule }
  | { type: 'many1'; item: Rule }
  | { type: 'optional'; item: Rule }
  | { type: 'ref'; rule: string }
  | { type: 'keyword'; value: string }
  | { type: 'literal'; value: string }
  | { type: 'identifier' }
  | { type: 'int' }
  | { type: 'float' }
  | { type: 'string' }
  | { type: 'block'; scope: string | null };

export interface RuleEntry {
  rule: Rule;
  handler?: string;
}

export interface DeclarationDef {
  keyword: string;
  nameRule: Rule;
  scopeRules: string[];
  inheritsBase: boolean;
  handler?: string;
}

export interface GrammarPackage {
  version: 1;
  package: string;
  keywords: string[];
  rules: Record<string, RuleEntry>;
  declarations: DeclarationDef[];
}

export function isGrammarPackage(value: unknown): value is GrammarPackage {
  if (value === null || typeof value !== 'object') return false;
  const obj = value as Record<string, unknown>;
  return (
    obj.version === 1 &&
    typeof obj.package === 'string' &&
    Array.isArray(obj.keywords) &&
    typeof obj.rules === 'object' && obj.rules !== null &&
    Array.isArray(obj.declarations)
  );
}

/** Merged grammar context from all loaded packages */
export interface GrammarContext {
  /** All declaration keywords across all packages */
  declarations: Map<string, DeclarationDef>;
  /** All rules across all packages (keyed by qualified name) */
  rules: Map<string, RuleEntry>;
  /** keyword → source package name */
  keywordSource: Map<string, string>;
  /** For each declaration keyword, the set of valid clause keywords in its scope */
  scopeKeywords: Map<string, Set<string>>;
}
```

```ts
import * as fs from 'fs/promises';
import * as path from 'path';
import * as TOML from '@iarna/toml';
import { GrammarPackage, GrammarContext, isGrammarPackage, RuleEntry } from './types';

interface Manifest {
  package: { name: string };
  dependencies?: Record<string, string>;
}

/** Try to read and parse a JSON file. Returns null if not found. */
async function readJsonFile<T>(filePath: string): Promise<T | null> {
  try {
    const text = await fs.readFile(filePath, 'utf-8');
    return JSON.parse(text) as T;
  } catch {
    return null;
  }
}

/** Find grammar.ir.json for a dependency package, checking target-specific paths first. */
async function findGrammarIr(pkgDir: string): Promise<GrammarPackage | null> {
  // Check target-specific paths first (e.g., dist/paper/grammar.ir.json)
  const distDir = path.join(pkgDir, 'dist');
  try {
    const entries = await fs.readdir(distDir, { withFileTypes: true });
    for (const entry of entries) {
      if (entry.isDirectory()) {
        const targetPath = path.join(distDir, entry.name, 'grammar.ir.json');
        const pkg = await readJsonFile<unknown>(targetPath);
        if (pkg && isGrammarPackage(pkg)) return pkg;
      }
    }
  } catch { /* dist dir may not exist */ }

  // Fallback: dist/grammar.ir.json
  const defaultPath = path.join(distDir, 'grammar.ir.json');
  const pkg = await readJsonFile<unknown>(defaultPath);
  if (pkg && isGrammarPackage(pkg)) return pkg;

  return null;
}

/** Extract clause keywords from a rule by collecting all keyword values. */
function extractKeywords(rules: Map<string, RuleEntry>, scopeRuleIds: string[]): Set<string> {
  const keywords = new Set<string>();
  for (const ruleId of scopeRuleIds) {
    const entry = rules.get(ruleId);
    if (entry) collectKeywordsFromRule(entry.rule, keywords);
  }
  return keywords;
}

function collectKeywordsFromRule(rule: { type: string; [key: string]: unknown }, set: Set<string>): void {
  if (rule.type === 'keyword') {
    set.add(rule.value as string);
  } else if (rule.type === 'seq' || rule.type === 'choice') {
    for (const item of rule.items as Array<{ type: string; [key: string]: unknown }>) {
      collectKeywordsFromRule(item, set);
    }
  } else if (rule.type === 'many' || rule.type === 'many1' || rule.type === 'optional') {
    collectKeywordsFromRule(rule.item as { type: string; [key: string]: unknown }, set);
  }
}

function mergePackage(pkg: GrammarPackage, ctx: GrammarContext): void {
  // Add rules
  for (const [id, entry] of Object.entries(pkg.rules)) {
    ctx.rules.set(id, entry);
  }

  // Add keywords
  for (const kw of pkg.keywords) {
    ctx.keywordSource.set(kw, pkg.package);
  }

  // Add declarations and compute scope keywords
  for (const decl of pkg.declarations) {
    ctx.declarations.set(decl.keyword, decl);
    ctx.scopeKeywords.set(decl.keyword, extractKeywords(ctx.rules, decl.scopeRules));
  }
}

export async function loadGrammarContext(workspaceRoot: string): Promise<GrammarContext> {
  const ctx: GrammarContext = {
    declarations: new Map(),
    rules: new Map(),
    keywordSource: new Map(),
    scopeKeywords: new Map(),
  };

  // 1. Read ink-package.toml
  const tomlPath = path.join(workspaceRoot, 'ink-package.toml');
  let manifest: Manifest;
  try {
    const tomlText = await fs.readFile(tomlPath, 'utf-8');
    manifest = TOML.parse(tomlText) as unknown as Manifest;
  } catch {
    return ctx; // No manifest — return empty context
  }

  // 2. Load dependency grammars
  const deps = manifest.dependencies ?? {};
  for (const depName of Object.keys(deps)) {
    const pkgDir = path.join(workspaceRoot, 'packages', depName);
    const pkg = await findGrammarIr(pkgDir);
    if (pkg) mergePackage(pkg, ctx);
  }

  // 4. Load project's own grammar from dist/
  const projectPkg = await findGrammarIr(workspaceRoot);
  if (projectPkg) mergePackage(projectPkg, ctx);

  return ctx;
}
```

```ts
import type { GrammarContext } from './types';

export enum TokenType {
  Keyword = 'keyword',
  Declaration = 'declaration',
  Clause = 'clause',
  Identifier = 'identifier',
  String = 'string',
  Number = 'number',
  Operator = 'operator',
  Comment = 'comment',
  Using = 'using',
}

export interface Token {
  type: TokenType;
  text: string;
  line: number;
  start: number;
  end: number;
  scope: string[];
}

export interface ParseResult {
  tokens: Token[];
  imports: string[];
  /** For each line, the scope stack at the start of that line */
  lineScopes: string[][];
}

const BASE_KEYWORDS = new Set([
  'let', 'fn', 'return', 'if', 'else', 'for', 'while', 'in',
  'break', 'continue', 'class', 'self', 'null', 'true', 'false', 'throw',
]);

const BUILTINS = new Set([
  'print', 'log', 'java', 'Math', 'io', 'json', 'db',
]);

const OPERATOR_RE = /^(==|!=|<=|>=|&&|\|\||[+\-*/%<>=!])/;
const NUMBER_RE = /^\d+(\.\d+)?/;
const IDENTIFIER_RE = /^[a-zA-Z_][\w.]*/;
const STRING_RE = /^"([^"\\]|\\.)*"/;
const UNCLOSED_STRING_RE = /^"([^"\\]|\\.)*/;

export function parseDocument(text: string, ctx: GrammarContext): ParseResult {
  const tokens: Token[] = [];
  const imports: string[] = [];
  const lines = text.split('\n');
  const scopeStack: string[] = [];
  const lineScopes: string[][] = [];
  let pendingScope: string | null = null; // Deferred scope push until `{` is found

  for (let lineNum = 0; lineNum < lines.length; lineNum++) {
    lineScopes.push([...scopeStack]);
    const line = lines[lineNum];
    let pos = 0;

    while (pos < line.length) {
      // Skip whitespace
      if (/\s/.test(line[pos])) { pos++; continue; }

      // Comment
      if (line[pos] === '/' && line[pos + 1] === '/') {
        tokens.push({ type: TokenType.Comment, text: line.slice(pos), line: lineNum, start: pos, end: line.length, scope: [...scopeStack] });
        break;
      }

      // String (closed or unclosed — consume rest of line if no closing quote)
      if (line[pos] === '"') {
        const strMatch = line.slice(pos).match(STRING_RE);
        if (strMatch) {
          tokens.push({ type: TokenType.String, text: strMatch[0], line: lineNum, start: pos, end: pos + strMatch[0].length, scope: [...scopeStack] });
          pos += strMatch[0].length;
          continue;
        }
        // Unclosed string — consume to end of line
        const unclosed = line.slice(pos);
        tokens.push({ type: TokenType.String, text: unclosed, line: lineNum, start: pos, end: line.length, scope: [...scopeStack] });
        break;
      }

      // Number
      const numMatch = line.slice(pos).match(NUMBER_RE);
      if (numMatch && (pos === 0 || !/[a-zA-Z_]/.test(line[pos - 1]))) {
        tokens.push({ type: TokenType.Number, text: numMatch[0], line: lineNum, start: pos, end: pos + numMatch[0].length, scope: [...scopeStack] });
        pos += numMatch[0].length;
        continue;
      }

      // Braces — scope tracking
      if (line[pos] === '{') {
        if (pendingScope) {
          scopeStack.push(pendingScope);
          pendingScope = null;
        }
        pos++;
        continue;
      }
      if (line[pos] === '}') {
        pendingScope = null; // Cancel any pending scope
        if (scopeStack.length > 0) scopeStack.pop();
        pos++;
        continue;
      }

      // Operator
      const opMatch = line.slice(pos).match(OPERATOR_RE);
      if (opMatch) {
        tokens.push({ type: TokenType.Operator, text: opMatch[0], line: lineNum, start: pos, end: pos + opMatch[0].length, scope: [...scopeStack] });
        pos += opMatch[0].length;
        continue;
      }

      // Identifier / keyword
      const idMatch = line.slice(pos).match(IDENTIFIER_RE);
      if (idMatch) {
        const word = idMatch[0];
        const tokenStart = pos;
        pos += word.length;

        // Using statement
        if (word === 'using') {
          tokens.push({ type: TokenType.Using, text: word, line: lineNum, start: tokenStart, end: pos, scope: [...scopeStack] });
          // Consume the package name
          const wsLen = line.slice(pos).length - line.slice(pos).trimStart().length;
          const rest = line.slice(pos).trimStart();
          const pkgMatch = rest.match(IDENTIFIER_RE);
          if (pkgMatch) {
            imports.push(pkgMatch[0]);
            const pkgStart = pos + wsLen;
            tokens.push({ type: TokenType.Identifier, text: pkgMatch[0], line: lineNum, start: pkgStart, end: pkgStart + pkgMatch[0].length, scope: [...scopeStack] });
            pos = pkgStart + pkgMatch[0].length;
          }
          continue;
        }

        // Declaration keyword (from grammar packages)
        if (ctx.declarations.has(word)) {
          tokens.push({ type: TokenType.Declaration, text: word, line: lineNum, start: tokenStart, end: pos, scope: [...scopeStack] });
          // Look ahead for name
          const whitespace = line.slice(pos).length - line.slice(pos).trimStart().length;
          const rest = line.slice(pos).trimStart();
          const nameMatch = rest.match(IDENTIFIER_RE);
          if (nameMatch) {
            const nameStart = pos + whitespace;
            tokens.push({ type: TokenType.Identifier, text: nameMatch[0], line: lineNum, start: nameStart, end: nameStart + nameMatch[0].length, scope: [...scopeStack] });
            pos = nameStart + nameMatch[0].length;
          }
          // Set pending scope — will be pushed when `{` is encountered
          pendingScope = `declaration:${word}`;
          continue;
        }

        // Clause keyword (valid in current scope)
        const currentDecl = scopeStack.findLast(s => s.startsWith('declaration:'))?.split(':')[1];
        if (currentDecl && ctx.scopeKeywords.get(currentDecl)?.has(word)) {
          tokens.push({ type: TokenType.Clause, text: word, line: lineNum, start: tokenStart, end: pos, scope: [...scopeStack] });
          // Set pending scope — will be pushed when `{` is encountered
          pendingScope = `clause:${word}`;
          continue;
        }

        // Base keyword
        if (BASE_KEYWORDS.has(word) || BUILTINS.has(word)) {
          tokens.push({ type: TokenType.Keyword, text: word, line: lineNum, start: tokenStart, end: pos, scope: [...scopeStack] });
          continue;
        }

        // Regular identifier
        tokens.push({ type: TokenType.Identifier, text: word, line: lineNum, start: tokenStart, end: pos, scope: [...scopeStack] });
        continue;
      }

      // Skip any other character (parens, brackets, commas, etc.)
      pos++;
    }
  }

  return { tokens, imports, lineScopes };
}
```

```ts
import { TokenType, type ParseResult } from '../parser';

// LSP semantic token types legend
export const TOKEN_TYPES = ['type', 'keyword', 'property', 'variable', 'string', 'number', 'comment'];
export const TOKEN_MODIFIERS = ['declaration', 'definition'];

/** Map parser token types to semantic token type indices */
function tokenTypeIndex(type: TokenType): number | null {
  switch (type) {
    case TokenType.Declaration: return TOKEN_TYPES.indexOf('type');
    case TokenType.Clause: return TOKEN_TYPES.indexOf('keyword');
    default: return null; // Only highlight grammar-package tokens semantically
  }
}

/**
 * Compute encoded semantic tokens from a parse result.
 * Returns a flat array of [deltaLine, deltaStart, length, tokenType, modifiers].
 */
export function computeSemanticTokens(parsed: ParseResult): number[] {
  const data: number[] = [];
  let prevLine = 0;
  let prevStart = 0;

  // Filter to only tokens that get semantic highlighting
  const semanticTokens = parsed.tokens.filter(t =>
    t.type === TokenType.Declaration || t.type === TokenType.Clause
  );

  // Sort by line then start position
  semanticTokens.sort((a, b) => a.line - b.line || a.start - b.start);

  for (const token of semanticTokens) {
    const typeIdx = tokenTypeIndex(token.type);
    if (typeIdx === null) continue;

    const deltaLine = token.line - prevLine;
    const deltaStart = deltaLine === 0 ? token.start - prevStart : token.start;

    data.push(deltaLine, deltaStart, token.text.length, typeIdx, 0);

    prevLine = token.line;
    prevStart = token.start;
  }

  return data;
}
```

```ts
import { TokenType, type ParseResult } from '../parser';
import type { GrammarContext } from '../types';

export interface Diagnostic {
  line: number;
  start: number;
  end: number;
  message: string;
  severity: 'error' | 'warning';
}

export function computeDiagnostics(parsed: ParseResult, ctx: GrammarContext): Diagnostic[] {
  const diagnostics: Diagnostic[] = [];

  // 1. Unknown using packages
  const knownPackages = new Set(ctx.keywordSource.values());
  for (const imp of parsed.imports) {
    if (!knownPackages.has(imp)) {
      const token = parsed.tokens.find(t => t.type === TokenType.Identifier && t.text === imp);
      if (token) {
        diagnostics.push({
          line: token.line,
          start: token.start,
          end: token.end,
          message: `Unknown package "${imp}" — not found in installed dependencies`,
          severity: 'warning',
        });
      }
    }
  }

  // 2. Clause keyword in wrong declaration scope
  // Collect all clause keywords across all declarations
  const allClauseKeywords = new Map<string, string[]>(); // keyword → valid declaration types
  for (const [declKeyword, clauseSet] of ctx.scopeKeywords) {
    for (const clause of clauseSet) {
      if (!allClauseKeywords.has(clause)) allClauseKeywords.set(clause, []);
      allClauseKeywords.get(clause)!.push(declKeyword);
    }
  }
  // Check identifiers that match a clause keyword but are in the wrong scope
  for (const token of parsed.tokens) {
    if (token.type !== TokenType.Identifier) continue;
    if (!allClauseKeywords.has(token.text)) continue;
    // This identifier matches a known clause keyword — check if it's in a valid scope
    const declScope = token.scope.findLast((s: string) => s.startsWith('declaration:'))?.split(':')[1];
    if (!declScope) continue; // Not inside a declaration — not a clause error
    const validDecls = allClauseKeywords.get(token.text)!;
    if (!validDecls.includes(declScope)) {
      diagnostics.push({
        line: token.line,
        start: token.start,
        end: token.end,
        message: `"${token.text}" is not a valid clause inside "${declScope}" (valid in: ${validDecls.join(', ')})`,
        severity: 'error',
      });
    }
  }

  // 3. Duplicate declaration names
  const declNames = new Map<string, { line: number; start: number; end: number }>();
  const declTokens = parsed.tokens.filter(t => t.type === TokenType.Declaration);
  for (const declToken of declTokens) {
    // The name is the next Identifier token on the same line
    const nameToken = parsed.tokens.find(
      t => t.type === TokenType.Identifier && t.line === declToken.line && t.start > declToken.end
    );
    if (nameToken) {
      const key = `${declToken.text}:${nameToken.text}`;
      if (declNames.has(key)) {
        diagnostics.push({
          line: nameToken.line,
          start: nameToken.start,
          end: nameToken.end,
          message: `Duplicate ${declToken.text} declaration: "${nameToken.text}" already declared`,
          severity: 'warning',
        });
      } else {
        declNames.set(key, { line: nameToken.line, start: nameToken.start, end: nameToken.end });
      }
    }
  }

  return diagnostics;
}
```

```ts
import type { ParseResult } from '../parser';
import type { GrammarContext } from '../types';

export interface CompletionItem {
  label: string;
  kind: 'keyword' | 'declaration' | 'clause' | 'package' | 'builtin';
  detail?: string;
}

const BASE_KEYWORDS = [
  'let', 'fn', 'return', 'if', 'else', 'for', 'while', 'in',
  'break', 'continue', 'class', 'throw',
];

const BUILTINS = [
  'print', 'log', 'java', 'java.call', 'java.new', 'java.toString',
  'java.getStatic', 'java.putStatic', 'java.type', 'java.cast',
  'Math', 'io', 'json', 'db',
];

export function computeCompletions(
  parsed: ParseResult,
  ctx: GrammarContext,
  line: number,
  character: number,
  documentText: string,
): CompletionItem[] {
  const items: CompletionItem[] = [];

  // Check if cursor is in the package-name position after "using"
  const lineText = documentText.split('\n')[line] ?? '';
  const beforeCursor = lineText.slice(0, character);
  if (/^\s*using\s+\S*$/.test(beforeCursor) || /^\s*using\s*$/.test(beforeCursor)) {
    const packages = new Set(ctx.keywordSource.values());
    for (const pkg of packages) {
      items.push({ label: pkg, kind: 'package', detail: 'package' });
    }
    return items;
  }

  // Determine scope at this line
  const scope = line < parsed.lineScopes.length ? parsed.lineScopes[line] : [];
  const currentDecl = scope.findLast((s: string) => s.startsWith('declaration:'))?.split(':')[1];
  const inClause = scope.some((s: string) => s.startsWith('clause:'));

  if (inClause) {
    for (const kw of BASE_KEYWORDS) {
      items.push({ label: kw, kind: 'keyword' });
    }
    for (const bi of BUILTINS) {
      items.push({ label: bi, kind: 'builtin' });
    }
  } else if (currentDecl) {
    const clauses = ctx.scopeKeywords.get(currentDecl);
    if (clauses) {
      for (const clause of clauses) {
        items.push({ label: clause, kind: 'clause', detail: `clause (${currentDecl})` });
      }
    }
    const decl = ctx.declarations.get(currentDecl);
    if (decl?.inheritsBase) {
      for (const kw of BASE_KEYWORDS) {
        items.push({ label: kw, kind: 'keyword' });
      }
    }
  } else {
    for (const [keyword] of ctx.declarations) {
      items.push({ label: keyword, kind: 'declaration', detail: `declaration (${ctx.keywordSource.get(keyword)})` });
    }
    for (const kw of BASE_KEYWORDS) {
      items.push({ label: kw, kind: 'keyword' });
    }
    items.push({ label: 'using', kind: 'keyword' });
  }

  return items;
}
```

```ts
import { TokenType, type ParseResult } from '../parser';
import type { GrammarContext, Rule } from '../types';

export interface HoverResult {
  contents: string;
  line: number;
  start: number;
  end: number;
}

/** Describe a rule structure in a human-readable way */
function describeRule(rule: Rule): string {
  switch (rule.type) {
    case 'keyword': return rule.value;
    case 'identifier': return '<name>';
    case 'string': return '<string>';
    case 'int': return '<int>';
    case 'float': return '<float>';
    case 'literal': return rule.value;
    case 'block': return '{ ... }';
    case 'ref': return `<${rule.rule}>`;
    case 'seq': return rule.items.map(describeRule).join(' ');
    case 'choice': return rule.items.map(describeRule).join(' | ');
    case 'many': return `(${describeRule(rule.item)})*`;
    case 'many1': return `(${describeRule(rule.item)})+`;
    case 'optional': return `(${describeRule(rule.item)})?`;
  }
}

export function computeHover(
  parsed: ParseResult,
  ctx: GrammarContext,
  line: number,
  character: number,
): HoverResult | null {
  // Find the token at this position
  const token = parsed.tokens.find(
    t => t.line === line && character >= t.start && character < t.end
  );
  if (!token) return null;

  if (token.type === TokenType.Declaration) {
    const decl = ctx.declarations.get(token.text);
    if (!decl) return null;
    const source = ctx.keywordSource.get(token.text);
    const clauses = ctx.scopeKeywords.get(token.text);
    const clauseList = clauses ? [...clauses].join(', ') : 'none';

    return {
      contents: `**${token.text}** declaration (from \`${source}\`)\n\nValid clauses: ${clauseList}`,
      line: token.line,
      start: token.start,
      end: token.end,
    };
  }

  if (token.type === TokenType.Clause) {
    const source = ctx.keywordSource.get(token.text);
    // Find the rule entry for this clause
    let syntax = '';
    for (const [, entry] of ctx.rules) {
      if (entry.rule.type === 'seq' && entry.rule.items.length > 0) {
        const first = entry.rule.items[0];
        if (first.type === 'keyword' && first.value === token.text) {
          syntax = describeRule(entry.rule);
          break;
        }
      }
    }

    return {
      contents: `**${token.text}** clause (from \`${source}\`)${syntax ? `\n\nSyntax: \`${syntax}\`` : ''}`,
      line: token.line,
      start: token.start,
      end: token.end,
    };
  }

  return null;
}
```

```ts
import { fileURLToPath } from 'url';
import {
  createConnection,
  InitializeParams,
  InitializeResult,
  ProposedFeatures,
  TextDocumentSyncKind,
  TextDocuments,
} from 'vscode-languageserver/node';
import { TextDocument } from 'vscode-languageserver-textdocument';
import { loadGrammarContext } from './grammar-loader';
import { parseDocument, type ParseResult } from './parser';
import { computeSemanticTokens, TOKEN_TYPES, TOKEN_MODIFIERS } from './features/semantic-tokens';
import { computeDiagnostics } from './features/diagnostics';
import { computeCompletions } from './features/completion';
import { computeHover } from './features/hover';
import type { GrammarContext } from './types';

const connection = createConnection(ProposedFeatures.all);
const documents = new TextDocuments(TextDocument);

let workspaceRoot: string | null = null;
let grammarCtx: GrammarContext = {
  declarations: new Map(),
  rules: new Map(),
  keywordSource: new Map(),
  scopeKeywords: new Map(),
};

// Cache parsed results per document URI
const parseCache = new Map<string, ParseResult>();

connection.onInitialize(async (params: InitializeParams): Promise<InitializeResult> => {
  workspaceRoot = params.rootUri
    ? fileURLToPath(params.rootUri)
    : params.rootPath ?? null;

  if (workspaceRoot) {
    try {
      grammarCtx = await loadGrammarContext(workspaceRoot);
    } catch (e) {
      connection.console.error(`Failed to load grammar context: ${e}`);
    }
  }

  return {
    capabilities: {
      textDocumentSync: TextDocumentSyncKind.Full,
      completionProvider: { triggerCharacters: [' '] },
      hoverProvider: true,
      semanticTokensProvider: {
        legend: { tokenTypes: TOKEN_TYPES, tokenModifiers: TOKEN_MODIFIERS },
        full: true,
      },
    },
  };
});

function refreshDocument(doc: TextDocument): void {
  const parsed = parseDocument(doc.getText(), grammarCtx);
  parseCache.set(doc.uri, parsed);

  // Publish diagnostics
  const diags = computeDiagnostics(parsed, grammarCtx);
  connection.sendDiagnostics({
    uri: doc.uri,
    diagnostics: diags.map(d => ({
      range: {
        start: { line: d.line, character: d.start },
        end: { line: d.line, character: d.end },
      },
      message: d.message,
      severity: d.severity === 'error' ? 1 : 2,
    })),
  });
}

documents.onDidOpen(e => refreshDocument(e.document));
documents.onDidChangeContent(e => refreshDocument(e.document));
documents.onDidClose(e => {
  parseCache.delete(e.document.uri);
  connection.sendDiagnostics({ uri: e.document.uri, diagnostics: [] });
});

connection.languages.semanticTokens.on((params) => {
  const parsed = parseCache.get(params.textDocument.uri);
  if (!parsed) return { data: [] };
  return { data: computeSemanticTokens(parsed) };
});

connection.onCompletion((params) => {
  const doc = documents.get(params.textDocument.uri);
  const parsed = parseCache.get(params.textDocument.uri);
  if (!parsed || !doc) return [];

  const items = computeCompletions(
    parsed, grammarCtx,
    params.position.line, params.position.character,
    doc.getText(),
  );

  return items.map(item => ({
    label: item.label,
    kind: item.kind === 'keyword' ? 14 : item.kind === 'declaration' ? 7 : item.kind === 'clause' ? 14 : item.kind === 'package' ? 9 : 3,
    detail: item.detail,
  }));
});

connection.onHover((params) => {
  const parsed = parseCache.get(params.textDocument.uri);
  if (!parsed) return null;

  const result = computeHover(parsed, grammarCtx, params.position.line, params.position.character);
  if (!result) return null;

  return {
    contents: { kind: 'markdown', value: result.contents },
    range: {
      start: { line: result.line, character: result.start },
      end: { line: result.line, character: result.end },
    },
  };
});

// File watch: reload grammars when config files change
connection.onDidChangeWatchedFiles(async () => {
  if (workspaceRoot) {
    try {
      grammarCtx = await loadGrammarContext(workspaceRoot);
      // Refresh all open documents
      for (const doc of documents.all()) {
        refreshDocument(doc);
      }
    } catch (e) {
      connection.console.error(`Failed to reload grammar context: ${e}`);
    }
  }
});

documents.listen(connection);
connection.listen();
```

# Ink Language Support

VS Code extension providing language support for the [Ink](https://github.com/inklang) programming language.

## Features

- **Syntax highlighting** — Base Ink keywords highlighted via TextMate grammar
